	value = 0
	every_in_list = {
		variable = accou_better_councillor_options
		save_temporary_scope_as = councillor
		min = scope:character.accou_councillor_skill_value
	}
}

	if = {
		limit = { scope:councillor = flag:chancellor }
		value = diplomacy

		limit = { scope:councillor = flag:steward }
		value = stewardship
	}
	else_if = {
		limit = { scope:councillor = flag:marshal }
		value = martial
	}
	else_if = {
		limit = { scope:councillor = flag:spymaster }
		value = intrigue
	}
	else_if = {
		limit = { scope:councillor = flag:court_chaplain }
		value = learning
	}
}

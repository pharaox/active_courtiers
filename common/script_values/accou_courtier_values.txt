accou_better_councillor_best_skill_value = {
	save_temporary_scope_as = character
	value = 0
	every_in_list = {
		variable = accou_better_councillor_options
		if = {
			limit = { this = flag:chancellor }
			min = scope:character.diplomacy
		}
		else_if = {
			limit = { this = flag:steward }
			min = scope:character.stewardship
		}
		else_if = {
			limit = { this = flag:marshal }
			min = scope:character.martial
		}
		else_if = {
			limit = { this = flag:spymaster }
			min = scope:character.intrigue
		}
		else_if = {
			limit = { this = flag:court_chaplain }
			min = scope:character.learning
		}
	}
}

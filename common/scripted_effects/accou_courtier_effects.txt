accou_get_min_apitude_court_position_holder_effect = {
	if = {
		limit = {
			any_court_position_holder = {
				type = $TYPE$
				count >= $COUNT$
			}
		}
		ordered_court_position_holder = {
			type = $TYPE$
			order_by = {
				value = 0
				subtract = aptitude:$TYPE$
			}
			save_scope_as = $HOLDER$
		}
	}
}
